## Description
There seems to be a bug with the **create2** operation: whenever I try to deploy any contract with **linked libraries**, the EVM reverts the operation.
I've double checked that the bytecode I'm deploying is correct; I just took it from the contract's bytecode deployed on rinkeby through truffle previously linking the library to the Smartcontract like so:
``` Javascript
 deployer.deploy(Migrations)
 deployer.deploy(Library)
 deployer.link(Library, Contract)
 deployer.deploy(Contract)
```
( https://rinkeby.etherscan.io/address/0xae5dd2a633e690f3c739352a4aba992cca96f55c#code )

Here is an example of one of the contracts & its respective library that I've trying to deploy via create2:

 * Library:
```solidity
pragma solidity ^0.5.0;

library Library {
   struct Data {
       bool boolean;
   }
   
   function set(Data storage self, bool newBoolean) public {
       self.boolean = newBoolean;
   }
}
```
* Contract using the above library:
```solidity
pragma solidity ^0.5.0;

import "./Library.sol";

contract Contract {
   using Library for Library.Data;
   
   Library.Data private Data;
   
   function set(bool newBoolean) public {
       Data.set(newBoolean);
   }
}
```
* Factory contract (Where create2 lives):
```solidity
 pragma solidity ^0.5.0;

contract Factory {
  function deploy(bytes memory code, uint256 salt) public {
    address addr;
    assembly {
      addr := create2(0, add(code, 0x20), mload(code), salt)
      if iszero(extcodesize(addr)) {
        revert(0, "Problem deploying contract")
      }
    }
  }
```
