It takes about 6 seconds for the following code to be compiled. Every additional 'if' clause roughly doubles this time.

pragma solidity ^0.4.7;

contract test {
    
    uint x0;
    function b() returns (uint) {
        uint answer;
        
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        if(x0>3) answer = 42;
        
        return answer;
    }
    
}
